#version 400 core

in vec2 pass_textureCoords;
in vec2 surfaceNormal;
in vec2 toLightVector;

out vec4 out_Color;

uniform sampler2D textureSampler;
uniform vec3 lightColor;
uniform float lightStrength;

void main(void){

	float brightness = 0.1 / length(toLightVector);
	
	out_Color = brightness * lightStrength * texture(textureSampler, pass_textureCoords);

}
